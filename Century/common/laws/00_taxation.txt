# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_no_poll_taxes = {
	group = lawgroup_taxation
	
	icon = "gfx/interface/icons/law_icons/no_poll_tax.dds"
	
	progressiveness = 0
	
	# AI shouldn't change to land based taxation from more advanced taxation forms
	ai_will_do = {
		OR = {
			AND = {
				has_law_or_variant = law_type:law_low_corporation_tax
				has_law_or_variant = law_type:law_moderate_income_tax
			}
			AND = {
				has_law_or_variant = law_type:law_moderate_corporation_tax
				has_law_or_variant = law_type:law_low_income_tax
			}
		}
	}
}
