je_chinese_missions = {
	icon = "gfx/interface/icons/religion_icons/protestant.dds"

	group = je_group_historical_content

	scripted_button = CHI_deport_missionaries_button

	is_shown_when_inactive = {
		exists = c:CHI
		THIS = c:CHI
	}

	possible = {
		custom_tooltip = {
			text = lost_opium_wars_tt
			has_global_variable = lost_opium_wars
		}
		any_neighbouring_state = {
			is_treaty_port = yes
			custom_tooltip = {
				text = owner_is_christian_tt
				owner = {
					religion = {
						has_discrimination_trait = christian
					}
				}
			}
		}
		#this is stupid
		#NOT = {
		#	has_law_or_variant = law_type:law_isolationism
		#}
	}

	immediate = {
		custom_tooltip = {
			text = missionary_activities_begin_tt
			trigger_event = {
				id = heavenly.1
				days = 1
			}
		}
		every_scope_state = {
			limit = {
				any_neighbouring_state = {
					is_treaty_port = yes
					owner = {
						religion = {
							has_discrimination_trait = christian
						}
					}
				}
			}
			set_variable = open_to_missionaries_state
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_law_or_variant = law_type:law_no_migration_controls
					any_scope_state = {
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
				}
				every_scope_state = {
					limit = {
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
					set_variable = open_to_missionaries_state
				}
			}
			else = {
				every_scope_state = {
					limit = {
						any_neighbouring_state = {
							is_treaty_port = yes
							owner = {
								religion = {
									has_discrimination_trait = christian
								}
							}
						}
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
					set_variable = open_to_missionaries_state
				}
			}
		}
		random_events = {
			500 = 0
			10 = CHI_missionaries.1
			10 = CHI_missionaries.2
			10 = CHI_missionaries.3
			10 = CHI_missionaries.4
			10 = CHI_missionaries.5
		}
	}

	complete = {
		custom_tooltip = {
			text = missionary_ban_tt
			has_variable = CHI_deported_missionaries
		}
		OR = {
			is_country_type = recognized
			has_law_or_variant = law_type:law_state_atheism
			has_law_or_variant = law_type:law_isolationism
			has_law_or_variant = law_type:law_closed_borders
		}
	}
	
	current_value = {
		value = var:missionary_works_counter
	}

	goal_add_value = {
		value = 5
	}
	
	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}

		}
	}

			any_neighbouring_state = {

					text = religion_christian_trait_tt

						religion = {
							has_discrimination_trait = christian
						}
					}
				}
			}
			is_capital = no
		}
		hidden_trigger = {
			NOT = { has_global_variable = taiping_je_happened }
		}
	}

	immediate = {
		set_global_variable = taiping_je_happened
		every_scope_state = {
			limit = {
				state_region = s:STATE_NANJING
			}
			add_modifier = {
				name = god_worshippers
				days = long_modifier_time
			}
		}
		trigger_event = {
			id = heavenly.2
		}
	}

	on_monthly_pulse = {
		random_events = {
			4 = 0
			1 = heavenly.3
		}
	}

	complete = {
		OR = {
			AND = {
				has_law_or_variant = law_type:law_total_separation
				literacy_rate > 0.4
			}
			NOT = {
				any_scope_state = {
					has_modifier = god_worshippers
				}
			}
		}
	}
	
	on_complete = {
		ig:ig_intelligentsia = { 
			add_ideology = ideology_anti_clerical
		}
	}

	fail = {
		custom_tooltip = {
			text = taiping_rebellion_organised_tt
			scope:journal_entry = {
				"scripted_bar_progress(taiping_organisation_progress_bar)" >= 9
			}
		}
	}

	on_fail = {
		trigger_event = {
			id = heavenly.4
		}
		ig:ig_intelligentsia = { 
			add_ideology = ideology_anti_clerical
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes
}

je_taiping = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	
	group = je_group_historical_content

	scripted_button = qing_accept_han
	scripted_button = qing_increase_recruitment
	scripted_button = qing_rescind_increase_recruitment
	scripted_button = taiping_interest_moratorium_war
	
	modifiers_while_active = {
		modifier_taiping_occupation
	}
	
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_variable = var_qing_increased_recruitment
				}
				remove_modifier = qing_delegate_levies
				add_modifier = {
					name = qing_delegate_levies
					multiplier = global_var:qing_delegate_levies_multiplier
				}
				every_interest_group = {
					limit = {
						OR = {
							is_interest_group_type = ig_landowners
							is_interest_group_type = ig_armed_forces
						}
					}
					hidden_effect = { remove_modifier = qing_provincial_political_power_increase }
					add_modifier = {
						name = qing_provincial_political_power_increase
						is_decaying = yes
						years = 40
						multiplier = global_var:qing_delegate_levies_multiplier
					}
				}
				change_global_variable = {
					name = qing_delegate_levies_multiplier
					add = 1
				}
				every_scope_war = {
					add_war_war_support = {
						target = ROOT
						value = 200
					}
				}
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = taiping_destroyed
			hidden_trigger = {
				NOT = {
					exists = c:TPG
				}
			}
		}
	}

	immediate = {
		c:TPG = {
			save_scope_as = taiping_scope
		}
		set_global_variable = {
			name = qing_delegate_levies_multiplier
			value = 1
		}
	}

	on_complete = {
		trigger_event = { id = heavenly.5 days = 0 }
		if = {
			limit = {
				has_modifier = qing_delegate_levies
			}
			remove_modifier = qing_delegate_levies
		}
		custom_tooltip = {
			text = interest_moratorium_edict_continue_tt
			if = {
				limit = {
					has_modifier = interest_moratorium_edict
				}
				remove_modifier = interest_moratorium_edict
				add_modifier = {
					name = interest_moratorium_edict
					years = 10
				}
			}
		}
		hidden_effect = {
			remove_modifier = taiping_taiping_rebellion
		}
		add_journal_entry = {
			type = je_taiping_reconstruction
		}
	}

	fail = {
		exists = c:TPG
		has_truce_with = c:TPG
	}

	on_fail = {
		hidden_effect = {
			remove_modifier = taiping_taiping_rebellion
			c:TPG = {
				remove_modifier = taiping_taiping_rebellion
			}
		}
	}

	weight = 1000
	should_be_pinned_by_default = yes

	transferable = no
}
