taiping_concurrent_rebellion_setup = {
	capital = {
		add_modifier = {
			name = modifier_taiping_hiring_bonus
			months = 6
		}
		save_scope_as = capital_scope
	}
	create_military_formation = {
		type = army
		hq_region =  scope:capital_scope.region

		combat_unit = {
			type = unit_type:combat_unit_type_irregular_infantry
			state_region = scope:capital_scope.state_region
			count = 30
		}
	}
	every_scope_state = {
		state_region = {
			add_claim = c:CHI
		}
	}	
}
